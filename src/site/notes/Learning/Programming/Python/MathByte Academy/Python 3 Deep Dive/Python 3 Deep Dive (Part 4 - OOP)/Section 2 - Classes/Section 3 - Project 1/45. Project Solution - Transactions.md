from datetime import datetime, timezone

class TimeZone:  
    def __init__(self, name, offset_hours, offset_minutes):  
        if name is None or len(name.strip()) == 0:  
            raise ValueError('Timezone name cannot be empty')  
  
        if not isinstance(offset_hours, numbers.Integral):  
            raise ValueError('offset_hours must be integer')  
  
        if not isinstance(offset_minutes, numbers.Integral):  
            raise ValueError('offset_minutes must be integer')  
  
        if -59 > offset_minutes > 59:  
            raise ValueError('offset_minutes must be between -59 and 59(inclusive)')  
  
        offset = timedelta(hours=offset_hours, minutes=offset_minutes)  
        if timedelta(hours=12, minutes=0) > offset > timedelta(hours=14, minutes=0):  
            raise ValueError('offset must be between -12:00 and +14:00 hours')  
        self._name = name  
        self._offset_hours =  offset_hours  
        self._offset_minutes =  offset_minutes  
        self._offset = offset  
  
    @property  
    def offset(self):  
        return self._offset  
  
    @property  
    def name(self):  
        return self._name  
  
    def __eq__(self, other):  
        return isinstance(other, TimeZone) and self.name == other.name and self._offset_hours == other._offset_hours and self._offset_minutes == other._offset_minutes  
  
    def __repr__(self):  
        return f"Timezone(name={self.name}, offset_hours={self._offset_hours}, offset_minutes={self._offset_minutes})"

Confirmation = namedtuple('Confirmation', 'account_number transaction_code transaction_id time_utc time')  
  
class BankAccount:  
    _interest_rate = 0.5 # percent  
    transaction_count = itertools.count(100)

    def validate_and_set_name(self, attr_name, value, field_tile):  
        if value is None or len(str(value).strip()) == 0:  
            raise ValueError(f'{field_tile} cannot be empty')  
        setattr(self, attr_name, value)  
  
    @property  
    def preferred_timezone(self):  
        return self._preferred_timezone  
  
    @preferred_timezone.setter  
    def preferred_timezone(self, value):  
        if not isinstance(value, TimeZone):  
            raise ValueError('Time zone must be a valid Timezone object')  
        self._preferred_timezone = value  
  
    @classmethod  
    def get_interest_rate(cls):  
        return cls._interest_rate  
  
    @classmethod  
    def set_interest_rate(cls, value):  
        if not isinstance(value, numbers.Real):  
            raise ValueError('Interest rate must be a real number')  
  
        if value < 0:  
            raise ValueError('Interest rate must be a positive number')  
        cls._interest_rate = value  
  
    @property  
    def balance(self):  
        return self._balance  
  
    @balance.setter  
    def balance(self, value):  
        if value < 0:  
            raise ValueError('Balance cannot be -ve')  
        self._balance = float(value)  
  
    @staticmethod  
    def validate_real_number(value, min_value):  
        if not isinstance(value, numbers.Real):  
            raise ValueError('Value must be a real number')  
        if min_value is not None and value < min_value:  
            raise ValueError(f'Value must be greater than or equal to {min_value}')  
        return value  
  
    def generate_confirmation_number(self, transaction_code):  
        utc_timestamp = datetime.now(timezone.utc).strftime('%Y%m%d%H%M%S')  
        return f"{transaction_code}-{self.account_number}-{utc_timestamp}-{next(BankAccount.transaction_count)}"  
  
  
    # def make_transaction(self):

    @staticmethod  
    def parse_confirmation(confirmation_number, p_timezone=None):  
        parts = confirmation_number.split('-')  
        if len(parts) != 4:  
            raise ValueError('Invalid confirmation code')  
        transaction_code, account_number, raw_dt_utc, transaction_id = parts  
        try:  
            dt_utc = datetime.strptime(raw_dt_utc, '%Y%m%d%H%M%S')  
        except ValueError as ex:  
            raise ValueError('Invalid transaction datetime') from ex  
        if p_timezone is None:  
            p_timezone = TimeZone('UTC', 0, 0)  
        if not isinstance(p_timezone, TimeZone):  
            raise ValueError('Invalid timezone specified')  
        p_dt = dt_utc + p_timezone.offset  
        p_dt_str = f"{p_dt.strftime('%Y-%m-%d %H:%M:%S')} ({p_timezone.name})"  
        return Confirmation(account_number, transaction_code, dt_utc, dt_utc.isoformat(), p_dt_str)  
  
    def deposit(self,  amount, transaction_code='D'):  
        amount = BankAccount.validate_real_number(amount, 0.01)  
        transaction_code = BankAccount._transaction_codes['deposit']  
        conf_code = self.generate_confirmation_number(transaction_code)  
        self._balance = self._balance + amount  
        return conf_code  
  
    def withdraw(self, amount):  
        amount = BankAccount.validate_real_number(amount, 0.01)  
        accepted = False  
        if amount < self.balance:  
            accepted = True  
            transaction_code = BankAccount._transaction_codes['withdraw']  
  
        else:  
            transaction_code = BankAccount._transaction_codes['rejected']  
        conf_code = self.generate_confirmation_number(transaction_code)  
        if accepted:  
            self.balance = self.balance - amount  
        return conf_code  
  
    def pay_interest(self):  
        transaction_code = BankAccount._transaction_codes['interest']  
        conf_code = self.generate_confirmation_number(transaction_code)  
        self.balance = self.balance + self.balance * (BankAccount.get_interest_rate() / 12)  
        return conf_code
